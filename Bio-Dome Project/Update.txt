

   if ((Temperature >= 55 && Temperature <= 75) && 
       (Humidity >= 50 && Humidity <= 70) &&
       (WindSpeed >= 0 && WindSpeed <= 10))
   { 
     //  Do Nothing
     stateChange = 1;
   }
   else
   { 
     if (Temp > 75)
     {
        TurnOffBioDomeHeater(bioDome);
        if (stateChange < 1)
        {
	       OpenBioDomeVent(bioDome);	
        }
        if (stateChange < 1)
        {
	       TurnOnBioDomeSprinkler(bioDome); 
        }
        if (stateChange < 1)
        {
	       // Throw an Error Message about no resources available
        }
      }
      else if (Temp < 55)
      {
        CloseBioDomeVent(bioDome);
        if (stateChange < 1)
        {
           TurnOffBioDomeSprinkler(bioDome)
        }
        if (stateChange < 1)
        {
           TurnOnBioDomeHeater(bioDome)
        }
        if (stateChange < 1)
        {
	       // Throw an Error Message about no resources available
        }
      }
      else if (Humid > 70)
      {
        TurnOffBioDomeSprinkler(bioDome)
        if (stateChange < 1)
        {
	       OpenBioDomeVent(bioDome);	
        }
        if (stateChange < 1) 
		{
	       // Throw an Error Message about no resources available
        }
      }
      else if (Humid < 50)
      {
        TurnOnBioDomeSprinkler(bioDome); 
        if (stateChange < 1)
        {
           CloseBioDomeVent(bioDome);
        }
        if (stateChange < 1)
        {
	       // Throw an Error Message about no resources available
        }
      }
      else if (Wind > 10)
      {
        CloseBioDomeVent(bioDome);

        {
